def create_user_dashboard_relationship(user, dashboard):
    """Create relationship between user and dashboard"""
    if isinstance(user, Customer):
        return CustomerDashboard(user)
    elif isinstance(user, Plumber):
        return PlumberDashboard(user)
    elif isinstance(user, Cleaner):
        return CleanerDashboard(user)
    elif isinstance(user, Electrician):
        return ElectricianDashboard(user)
    else:
        return None
